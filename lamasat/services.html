01

أعمال تصميم وتنفيذ الديكور الداخلي والخارجي

تقديم كافة أعمال التصميم من مناظير ومخططات تنفيذية وتوزيع الفرش والأثاث وحصر الكميات .بإسلوب مبدع وجذاب وتنفيذها بجودة وإتقان

02

الاعمال الكهربائية

نقدم كافة أعمال الكهرباء من مرحلة التأسيس إلى مرحلة التشطيب على حسب المخططات المرفقة للمشروع وذلك بجودة وضمانات على كافة الأعمال.

03

أعمال السباكة

نقــدم كافــة أعمــال الســباكة مــن مرحلــة التأســيس إلــى مرحلــة التشــطيب علــى حســب المخططــات المقدمـة فـي المشـروع وتنفيذهـا بجـودة مـع ضمـان كافــة األعمــال

04

أعمال العزل وااليبوكسي

نقــدم أعمــال العــزل مــع فريــق مختــص بأعمــال العـزل الحـراري والمائـي بأسـاليب وطـرق متنوعـة علـى حسـب طبيعـة العمـل وكذلـك نقـدم أعمـال ارضيـات ايبوكسـي للمصانـع والمسـتودعات وتنفيذهـا بجـودة عاليــة مــع ضمــان جــودة العمــل.

05

أعمال البلاط

تنفيــذ كافــة أعمــال البــاط مــن ســيراميك وبورســان ورخــام وانترلــوك بطــرق جيــدة وصحيحــة ومطابقــة لنــوع العمــل ســواء بــاط ارضيــات أو جــدران وبدقــة عاليــة مــع ضمــان األعمــال.

06

أعمال الدهانات

تنفيــذ أعمــال الدهــان بكافــة أنواعهــا مــع مراعــاة كل مرحلــة مــن مراحــل الدهــان وتنفيذهــا بشــكل المطلــوب باســلوب جيــد مــع فريــق عمــل متميــز ومبــدع وتقديــم الضمــان علــى جــودة األعمــال.
